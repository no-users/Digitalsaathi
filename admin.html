async function processCardPanelFormSubmission() { // 1. Sirf text inputs ka validation (Standard Fields) const requiredFields = [ 'cardMainFolderContextSelector', 'cardSubPortalContextSelector', 'cardTitleHeaderInput', 'cardUrlPathInput' ]; if (!validateFormFields(requiredFields)) return; // 2.
